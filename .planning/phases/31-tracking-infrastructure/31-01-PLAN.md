**Change 3: Replace inline script (lines 186-242)**
Replace the entire `<script is:inline>` block (the Google Ads gtag section) with the Consent Mode v2 implementation. The new script must follow this exact order:

```javascript
(function() {
  var GTAG_ID = 'AW-17409050017';
  var CONSENT_KEY = 'nettup_ads_consent';

  // 1. Define dataLayer and gtag function
  window.dataLayer = window.dataLayer || [];
  function gtag(){window.dataLayer.push(arguments);}
  window.gtag = gtag;

  // 2. Set consent defaults BEFORE loading gtag.js (CRITICAL ORDER)
  gtag('consent', 'default', {
    'ad_storage': 'denied',
    'ad_user_data': 'denied',
    'ad_personalization': 'denied',
    'analytics_storage': 'denied',
  });

  // 3. Check stored consent -- update immediately if previously granted
  var stored = localStorage.getItem(CONSENT_KEY);
  if (stored === 'granted') {
    gtag('consent', 'update', {
      'ad_storage': 'granted',
      'ad_user_data': 'granted',
      'ad_personalization': 'granted',
      'analytics_storage': 'granted',
    });
  }

  // 4. Load gtag.js (ALWAYS, regardless of consent state)
  var script = document.createElement('script');
  script.src = 'https://www.googletagmanager.com/gtag/js?id=' + GTAG_ID;
  script.async = true;
  document.head.appendChild(script);

  // 5. Initialize gtag
  gtag('js', new Date());
  gtag('config', GTAG_ID);
  window.gtagLoaded = true;

  // 6. Show banner only if no stored choice
  if (stored === null) {
    document.getElementById('cookie-banner')?.classList.remove('hidden');
  }

  // 7. Handle accept
  document.getElementById('cookie-accept')?.addEventListener('click', function() {
    localStorage.setItem(CONSENT_KEY, 'granted');
    gtag('consent', 'update', {
      'ad_storage': 'granted',
      'ad_user_data': 'granted',
      'ad_personalization': 'granted',
      'analytics_storage': 'granted',
    });
    hideBanner();
  });

  // 8. Handle decline
  document.getElementById('cookie-decline')?.addEventListener('click', function() {
    localStorage.setItem(CONSENT_KEY, 'denied');
    // No consent update needed -- defaults are already denied
    hideBanner();
  });

  function hideBanner() {
    var banner = document.getElementById('cookie-banner');
    if (banner) {
      banner.style.opacity = '0';
      banner.style.transform = 'translateY(100%)';
      banner.style.transition = 'opacity 0.3s, transform 0.3s';
      setTimeout(function() { banner.remove(); }, 300);
    }
  }
})();
```

Key differences from old script:
- gtag function defined and assigned to window.gtag FIRST (was only inside loadGtag)
- Consent defaults set BEFORE gtag.js loads (was missing entirely)
- All 4 consent params: ad_storage, ad_user_data, ad_personalization, analytics_storage (was missing v2 params)
- gtag.js ALWAYS loads (was conditional on consent)
- window.gtagLoaded = true always set (was only on consent)
- Returning visitor consent restored on page load (was only gtag loading)
- Use `var` not `const`/`let` inside is:inline scripts for broader compatibility
