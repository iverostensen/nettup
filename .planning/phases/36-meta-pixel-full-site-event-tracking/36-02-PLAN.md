**Step 1: Create test file** `src/lib/__tests__/utm.test.ts`:
```typescript
import { describe, it, expect, beforeEach, vi } from 'vitest';
import { captureUtmParams, getUtmParams } from '../utm';

// Mock sessionStorage
const store: Record<string, string> = {};
const mockSessionStorage = {
  getItem: vi.fn((key: string) => store[key] ?? null),
  setItem: vi.fn((key: string, value: string) => { store[key] = value; }),
  removeItem: vi.fn((key: string) => { delete store[key]; }),
  clear: vi.fn(() => { Object.keys(store).forEach(k => delete store[k]); }),
  get length() { return Object.keys(store).length; },
  key: vi.fn((i: number) => Object.keys(store)[i] ?? null),
};

describe('utm', () => {
  beforeEach(() => {
    mockSessionStorage.clear();
    vi.stubGlobal('sessionStorage', mockSessionStorage);
  });

  describe('captureUtmParams', () => {
    it('captures all 5 UTM params from URL', () => {
      const url = '?utm_source=facebook&utm_medium=cpc&utm_campaign=spring&utm_content=video1&utm_term=nettside';
      vi.stubGlobal('location', { search: url });
      captureUtmParams();
      expect(store['utm_source']).toBe('facebook');
      expect(store['utm_medium']).toBe('cpc');
      expect(store['utm_campaign']).toBe('spring');
      expect(store['utm_content']).toBe('video1');
      expect(store['utm_term']).toBe('nettside');
    });

    it('captures only present params', () => {
      vi.stubGlobal('location', { search: '?utm_source=facebook&utm_content=ad1' });
      captureUtmParams();
      expect(store['utm_source']).toBe('facebook');
      expect(store['utm_content']).toBe('ad1');
      expect(store['utm_medium']).toBeUndefined();
    });
  });

  describe('getUtmParams', () => {
    it('returns all stored UTM params', () => {
      store['utm_source'] = 'facebook';
      store['utm_medium'] = 'cpc';
      store['utm_campaign'] = 'spring';
      store['utm_content'] = 'video1';
      store['utm_term'] = 'nettside';
      const result = getUtmParams();
      expect(result).toEqual({
        utm_source: 'facebook',
        utm_medium: 'cpc',
        utm_campaign: 'spring',
        utm_content: 'video1',
        utm_term: 'nettside',
      });
    });

    it('omits params without stored values', () => {
      store['utm_source'] = 'google';
      const result = getUtmParams();
      expect(result).toEqual({ utm_source: 'google' });
      expect(result).not.toHaveProperty('utm_content');
      expect(result).not.toHaveProperty('utm_term');
    });
  });
});
```
