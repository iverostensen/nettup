**Form submit handler (per D-03, D-06, D-07):**
```typescript
const handleSubmit = async (e: React.FormEvent) => {
  e.preventDefault();
  if (!validate()) return;
  setStatus('submitting');
  try {
    const response = await fetch(`https://formspree.io/f/${SJEKKLISTE_FORMSPREE_ID}`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ navn, email, kilde: 'sjekkliste', ...getUtmParams() }),
    });
    if (response.ok) {
      setStatus('unlocked');
      trackLeadMagnetDownload();  // Plausible event (LEAD-02)
      if (localStorage.getItem('nettup_ads_consent') === 'granted' && window.fbq) {
        window.fbq('track', 'Lead', { content_name: 'Sjekkliste 2026' });
      }
    } else {
      setStatus('error');
    }
  } catch {
    setStatus('error');
  }
};
```
