  "@type": "Service",
  "name": `Nettup ${service.name}`,
  "description": service.description,
  "provider": {
    "@type": "Organization",
    "name": "Nettup",
    "url": "https://nettup.no"
  },
  "areaServed": {
    "@type": "Country",
    "name": "Norway"
  },
  "offers": {
    "@type": "Offer",
    "priceSpecification": {
      "@type": "PriceSpecification",
      "minPrice": service.minPrice,
      ...(service.maxPrice > 0 ? { "maxPrice": service.maxPrice } : {}),
      "priceCurrency": "NOK"
    }
  }
};

<BaseLayout
