**src/pages/blogg/[slug].astro** — Dynamic article page:

```astro
---
import { getCollection, render } from 'astro:content';
import type { CollectionEntry } from 'astro:content';
import BaseLayout from '@/layouts/BaseLayout.astro';
import Breadcrumbs from '@/components/ui/Breadcrumbs.astro';
import RelatedArticles from '@/components/blog/RelatedArticles.astro';

export async function getStaticPaths() {
  const articles = await getCollection('blogg');
  return articles.map((article) => ({
    params: { slug: article.slug },
    props: { article },
  }));
}

interface Props {
  article: CollectionEntry<'blogg'>;
}

const { article } = Astro.props;
const { Content } = await render(article);

const formattedDate = article.data.date.toLocaleDateString('nb-NO', {
  year: 'numeric', month: 'long', day: 'numeric',
});

// BlogPosting JSON-LD
const articleSchema = {
  "@context": "https://schema.org",
  "@type": "BlogPosting",
  "headline": article.data.seoTitle,
  "description": article.data.description,
  "datePublished": article.data.date.toISOString(),
  "dateModified": article.data.date.toISOString(),
  "url": `https://nettup.no/blogg/${article.slug}`,
  "mainEntityOfPage": {
    "@type": "WebPage",
    "@id": `https://nettup.no/blogg/${article.slug}`
  },
  "author": { "@type": "Organization", "name": "Nettup", "url": "https://nettup.no" },
  "publisher": {
    "@type": "Organization",
    "name": "Nettup",
    "url": "https://nettup.no",
    "logo": "https://nettup.no/images/nettup-logo.svg"
  },
  "inLanguage": "nb"
};

// BreadcrumbList JSON-LD — override BaseLayout's auto-generated one for article title accuracy
const breadcrumbSchema = {
  "@context": "https://schema.org",
  "@type": "BreadcrumbList",
  "itemListElement": [
    { "@type": "ListItem", "position": 1, "name": "Hjem", "item": "https://nettup.no/" },
    { "@type": "ListItem", "position": 2, "name": "Blogg", "item": "https://nettup.no/blogg" },
    { "@type": "ListItem", "position": 3, "name": article.data.title }
  ]
};

// FAQPage JSON-LD — only emit if article has faq frontmatter
const faqSchema = article.data.faq && article.data.faq.length > 0 ? {
  "@context": "https://schema.org",
  "@type": "FAQPage",
  "mainEntity": article.data.faq.map(item => ({
    "@type": "Question",
    "name": item.question,
    "acceptedAnswer": { "@type": "Answer", "text": item.answer }
  }))
} : null;
---

<BaseLayout title={article.data.seoTitle} description={article.data.description}>
  <Fragment slot="head">
    <meta property="og:type" content="article" />
    <meta property="article:published_time" content={article.data.date.toISOString()} />
    <meta property="article:section" content={article.data.category} />
    <script type="application/ld+json" set:html={JSON.stringify(articleSchema)} />
    <script type="application/ld+json" set:html={JSON.stringify(breadcrumbSchema)} />
    {faqSchema && <script type="application/ld+json" set:html={JSON.stringify(faqSchema)} />}
  </Fragment>

  <article class="mx-auto max-w-[70ch] px-4 py-16 md:py-24">
    <!-- Breadcrumb nav -->
    <Breadcrumbs items={[
      { label: 'Hjem', href: '/' },
      { label: 'Blogg', href: '/blogg' },
      { label: article.data.title },
    ]} />

    <!-- Article header -->
    <header class="mt-6 mb-10">
      <p class="mb-3 text-xs font-medium uppercase tracking-wider text-brand">{article.data.category}</p>
      <h1 class="font-display text-3xl font-semibold leading-tight md:text-4xl">{article.data.title}</h1>
      <p class="mt-4 text-sm text-text-muted">
        {formattedDate} · {article.data.readTime} min lesing
      </p>
    </header>

    <!-- Article body -->
    <div class="prose prose-invert prose-brand max-w-none">
      <Content />
    </div>

    <!-- CTA -->
    <div class="mt-16 rounded-md border border-brand/20 bg-surface-raised p-8 text-center">
      <p class="text-lg font-semibold">Trenger du en nettside?</p>
      <p class="mt-2 text-text-muted">Vi hjelper norske bedrifter med moderne, raske nettsider.</p>
      <a
        href="/kontakt"
        class="mt-6 inline-block rounded-md bg-brand px-6 py-3 text-sm font-semibold text-surface transition-colors hover:bg-brand-light"
      >
        Ta kontakt
      </a>
    </div>
  </article>

  <!-- Related articles -->
  <div class="mx-auto max-w-5xl px-4 pb-16">
    <RelatedArticles relatedSlugs={article.data.relatedSlugs} currentSlug={article.slug} />
  </div>
</BaseLayout>
```

## Hva påvirker prisen på en nettside?

Prisen på en nettside varierer enormt — fra 2 500 kr for en enkel mal til over 100 000 kr for skreddersydde løsninger. Her er de viktigste faktorene.

## Enkle nettsider (2 500 – 15 000 kr)

Passer for småbedrifter som trenger en profesjonell tilstedeværelse på nett. Inkluderer typisk 3-5 sider, kontaktskjema og mobiloptimalisering.

Hos Nettup starter våre nettsider på 2 500 kr i lanseringsperioden. Vi bruker moderne teknologi som gir rask lastetid og god SEO uten at du betaler for unødige funksjoner.

## Mellomstore nettsider (15 000 – 50 000 kr)

Inkluderer mer tilpasning, integrasjoner mot bookingsystemer, betalingsløsninger eller medlemsportaler. Les mer om [nettbutikk vs nettside](/blogg/nettbutikk-vs-nettside) hvis du er usikker på hva du trenger.

## Hva bør du alltid inkludere?

- SSL-sertifikat (HTTPS)
- Mobiloptimalisering
- Grunnleggende [SEO for småbedrifter](/blogg/seo-for-smabedrifter)
- Kontaktskjema med bekreftelsesmelding

## Vanlige spørsmål

**Hva koster en enkel nettside i Norge?**
En enkel bedriftsnettside koster typisk mellom 2 500 og 15 000 kr avhengig av leverandør og omfang.

**Er det månedlige kostnader?**
Ja, du betaler som regel for domene (150–300 kr/år) og hosting (50–300 kr/mnd).
