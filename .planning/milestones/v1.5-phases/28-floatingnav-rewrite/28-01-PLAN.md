<objective>
Convert FloatingNav from a `client:only` React island to a server-rendered Astro component with `transition:persist`, eliminating the React hydration gap that causes raw HTML flash on every SPA navigation. MobileMenu stays as a React island but becomes self-managing via a custom event instead of React state passed from FloatingNav.

Purpose: The current `client:only` FloatingNav produces an empty `<astro-island>` shell in SSR output. After every page navigation, the body becomes visible before React mounts, causing a brief flash of the page without the nav. The Astro component is in the DOM from the very first byte — no async mount, no flash.

Output:
- `src/components/layout/FloatingNav.astro` — server-rendered nav with `transition:persist`
- `src/components/islands/MobileMenu.tsx` — refactored to self-manage open state via custom event
- `src/layouts/BaseLayout.astro` — uses new Astro FloatingNav, removes `visibility:hidden` hack
- `src/styles/global.css` — adds nav CSS transitions/animations, removes `html[data-loading]` rule
</objective>

Frontmatter:
```astro

import { services } from '@/config/services';
import MobileMenu from '../islands/MobileMenu';

const navItems = [
  { name: 'Hjem', href: '/' },
  { name: 'Tjenester', href: '/tjenester' },
  { name: 'Om oss', href: '/om-oss' },
  { name: 'Prosjekter', href: '/prosjekter' },

const currentPath = Astro.url.pathname;

// Tjenester sub-page label logic (mirrors FloatingNav.tsx)
const tjenesterActive = currentPath.startsWith('/tjenester');
const currentServiceSlug = currentPath.split('/')[2];
const currentService = services.find(s => s.slug === currentServiceSlug);
const tjenesterLabel =
  tjenesterActive && currentPath !== '/tjenester' && currentService
    ? currentService.name
    : 'Tjenester';

function isActive(href: string): boolean {
  if (href === '/tjenester') return tjenesterActive;
  return currentPath === href;
}

const displayNavItems = navItems.map(item =>
  item.href === '/tjenester' ? { ...item, name: tjenesterLabel } : item
);

<script>
  let lastScrollY = 0;

  function initScrollBehavior() {
    const nav = document.getElementById('floating-nav');
    if (!nav) return;
    lastScrollY = window.scrollY;

    function onScroll() {
      const current = window.scrollY;
      if (current < 80) {
        nav!.classList.remove('nav-hidden');
      } else if (current > lastScrollY) {
        nav!.classList.add('nav-hidden');
      } else {
        nav!.classList.remove('nav-hidden');
      }
      lastScrollY = current;
    }

    window.addEventListener('scroll', onScroll, { passive: true });
  }

  function initEntranceAnimation() {
    const nav = document.getElementById('floating-nav');
    if (!nav) return;
    if (!sessionStorage.getItem('nav-animated')) {
      nav.classList.add('nav-entrance');
      sessionStorage.setItem('nav-animated', '1');
      nav.addEventListener('animationend', () => {
        nav.classList.remove('nav-entrance');
      }, { once: true });
    }
  }

  function updateActiveLinks() {
    const path = window.location.pathname;
    const tjenesterActive = path.startsWith('/tjenester');

    document.querySelectorAll<HTMLAnchorElement>('[data-nav-href]').forEach(link => {
      const href = link.dataset.navHref!;
      const active = href === '/tjenester' ? tjenesterActive : path === href;
      link.classList.toggle('text-brand', active);
      link.classList.toggle('text-text-muted', !active);
      link.classList.toggle('hover:text-text', !active);
      if (active) {
        link.setAttribute('aria-current', 'page');
      } else {
        link.removeAttribute('aria-current');
      }
    });
  }

  function initHamburger() {
    const btn = document.getElementById('hamburger-btn');
    if (!btn) return;
    btn.addEventListener('click', () => {
      document.dispatchEvent(new CustomEvent('open-mobile-menu'));
    });
  }

  // First load
  initScrollBehavior();
  initEntranceAnimation();
  initHamburger();

  // After each SPA navigation — update active links only.
  // Scroll and hamburger listeners are attached to window/btn which persist.
  document.addEventListener('astro:after-swap', updateActiveLinks);
</script>
```

This is an acceptable tradeoff — the color state is correct. Document in SUMMARY.
