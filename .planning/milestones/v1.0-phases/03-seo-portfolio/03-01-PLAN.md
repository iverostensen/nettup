In the BaseLayout.astro frontmatter (after the `const canonicalURL` line), compute the BreadcrumbList:

```typescript
const pageLabels: Record<string, string> = {
  '/': 'Hjem',
  '/tjenester': 'Tjenester',
  '/om-oss': 'Om oss',
  '/prosjekter': 'Prosjekter',
  '/kontakt': 'Kontakt',
};

const pathSegments = Astro.url.pathname.replace(/\/$/, '').split('/').filter(Boolean);
const breadcrumbItems: Array<{ position: number; name: string; item?: string }> = [
  { position: 1, name: 'Hjem', item: 'https://nettup.no/' },
  ...pathSegments.map((seg, i) => {
    const fullPath = '/' + pathSegments.slice(0, i + 1).join('/');
    return {
      position: i + 2,
      name: pageLabels[fullPath] ?? seg,
      item: `https://nettup.no${fullPath}`,
    };
  }),
];
// Omit `item` from the last entry per Google specification
if (breadcrumbItems.length > 1) {
  delete breadcrumbItems[breadcrumbItems.length - 1].item;
}

const breadcrumbSchema = {
  "@context": "https://schema.org",
  "@type": "BreadcrumbList",
  "itemListElement": breadcrumbItems.map(b => ({
    "@type": "ListItem",
    ...b,
  })),
};
```

In the `<head>` section, after the existing three JSON-LD blocks (Organization, LocalBusiness, Offer), add:

```astro
<!-- Structured Data: BreadcrumbList -->
<script type="application/ld+json" set:html={JSON.stringify(breadcrumbSchema)} />
```

IMPORTANT: Homepage (pathname = "/") produces a single-item breadcrumb with only "Hjem" — that single item retains its `item` URL (the delete only fires when length > 1). This is correct per Google spec.
